SDK · pip install agentloom (4 lines)

from agentloom import AgentLoom, MCP, A2A

# 1) 자사 SaaS의 백엔드에 직접 임베드
loom = AgentLoom(model="gemini-2.5-pro")

loom.use_mcp(MCP.stripe())          # 자사 결제 데이터
loom.use_mcp(MCP.postgres(my_db))     # 자사 DB
loom.connect_a2a("hr-agent.acme.com") # 사내 다른 agent

# 2) 자사 API 핸들러에서 호출
@app.post("/api/refund")
async def refund(req):
    result = await loom.run(
        agents=["intake", "policy", "compute", "audit"],
        task=req.body,
    )
    return result.decision
